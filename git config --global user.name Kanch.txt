### after creating a new branch online, you need to fetch it on git bash, then only you can see it on git bash
git fetch --all
git branch -r  -------to see all the branches

### to delete a remote branch on terminal, we need to make a push
git push -d origin Created-online

### to delete a branch locally(this will not delete from remote that means not delete from git repo)
git push -d  test-branch

git merge branch-1
